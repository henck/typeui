# Input: Theming 
.

The `Input` control takes theme information from a `ThemeProvider` that 
surrounds it. 

<Meta 
  title="Controls/Input/Theming" 
  component={Input}
  argTypes={{
    className: { table: { disable: true } }
  }}
/>

export const Template = (args) => 
<div style={{height: '600px', padding: '30px', background: '#406A8C'}}>
  <ThemeProvider theme={{...Theme, normalColor: '#aaa', background: '#305A7C', fontColor: '#fff' }}>
    <Form data={{}} onChange={() => {}} onValidate={() => {}}>
      <Form.Field
        name="text"
        value={null}
        control={<Input type="text" placeholder="Text" {...args}/>}
      />
      <Form.Field
        name="color"
        value={null}
        control={<Input type="color" placeholder="Color" {...args}/>}
      />
      <Form.Field
        name="date"
        value={null}
        control={<Input type="date" placeholder="Date" {...args}/>}
      />
      <Form.Field
        name="time"
        value={null}
        control={<Input type="time" clock placeholder="Time" {...args}/>}
      />
    </Form>
  </ThemeProvider>
</div>
