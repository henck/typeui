## Icon

A `Message` can contain an `Icon`. Adding the `icon` attribute will apply 
appropriate layout to the message.

export const IconTemplate = (args) =>
<Message {...args}>
  <Icon name="circle-notch" size="big" loading="true"/>
  <Message.Content>
    <Message.Header>Just one second</Message.Header>
    <p>
      We are fetching content for you.
    </p>
  </Message.Content>
</Message>

<Canvas>
  <Story
    name="Icon"
    args={{ icon: true }}>
    {IconTemplate.bind({})}
  </Story>
</Canvas>
