export const Template = (args) => <CircularProgress {...args}/>

## value

The `value` property is required. This must be a value between `0` and `100`.

export const ValueTemplate = (args) => <div>
  <CircularProgress {...args}/>
  <CircularProgress padded color="green" background value={50}/>
  <CircularProgress padded color="pink" background value={80}/>
  <CircularProgress padded color="red" background value={100}/>
</div>

<Canvas>
  <Story
    name="value"
    args={{value: 30,  onClick: null, padded: true, background: true}}>
    {ValueTemplate.bind({})}
  </Story>
</Canvas>

## onClick

If an `onClick` event handler is set, then the `CircularProgress` is clickable.

<Canvas>
  <Story
    name="onClick"
    args={{value: 30,  onClick: () => {} }}>
    {Template.bind({})}
  </Story>
</Canvas>
