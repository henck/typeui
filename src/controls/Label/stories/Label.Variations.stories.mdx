## basic

A `basic` Label has a border and a white background. Hovering labels with `onClick` only affects labels with default color. 

export const BasicTemplate = (args) =>
<div>
  <Label {...args}/>
  <Label basic color="teal" onClick={() => {}}>One</Label>
  <Label basic color="rebeccapurple" onClick={() => {}}><Image src='https://react.semantic-ui.com/images/avatar/small/stevie.jpg' />Three</Label>
</div>

<Canvas>
  <Story
    name="basic"
    args={{ basic: true, onClick: () => {}, children: "One" }}>
    {BasicTemplate.bind({})}
  </Story>
</Canvas>

## pointing 

A label can be `pointing` to `top`, `bottom`, `left` or `right`.

export const PointingTemplate = (args) => 
<div>
  <Input name="firstname" type="text" placeholder="First name" fluid/>
  <Label {...args}/>
  <Divider hidden/>
  <Label pointing="bottom">Please enter a value</Label>
  <Input name="lastname" type="text" placeholder="Last name" fluid/>
  <Divider hidden/>
  <Input name="username" type="text" placeholder="Username"/>
  <Label pointing>That name is taken!</Label>
  <Divider hidden/>
  <Label pointing="right">Your password must be 6 characters or more</Label>
  <Input name="password" type="password" placeholder="Password"/>
</div>

## tag 

A `Label` can appear as a `tag`. Tagged labels are meant to be placed on the right side of anything they're attached to.

export const TagTemplate = (args) =>
<div>
  <Label tag onClick={() => { alert('Clicked');}}>One</Label>
  <Label basic tag color="cornflowerblue" onClick={() => { alert('Clicked');}}>One</Label>
  <Label tag color="rgb(0, 156, 149)" onClick={() => { alert('Clicked');}}>Two</Label>
  <Label tag color="rebeccapurple" onClick={() => { alert('Clicked');}}><Image src='https://react.semantic-ui.com/images/avatar/small/stevie.jpg' />Three</Label>
</div>
